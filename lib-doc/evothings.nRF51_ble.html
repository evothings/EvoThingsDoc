evothings.nRF51_ble

Source: evothings-libraries/libs/evothings/nordic-nRF51-ble/nordic-nRF51-ble.js:26

Functions for communicating with a Nordic BLE device.

Example 

evothings.nRF51_ble.connect(
	'nRF51-DK', // BLE name
	function(device)
	{
		console.log('connected!');
		device.writeDataArray(new Uint8Array([1]));
		evothings.nRF51_ble.close();
	},
	function(errorCode)
	{
		console.log('Error: ' + errorCode);
	});
